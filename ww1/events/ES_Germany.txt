		name = germany_es.2.a

			factor = 90

				factor = 0

			country_event = { days = 1 id = germany.22 }
		}
	}
	option = { # Ask our faction leader
		name = germany_es.2.b
		trigger = {
			AND = {
				is_in_faction = yes
				is_faction_leader = no
			}
		}
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0
				OR = {
					is_in_faction = no
					is_faction_leader = yes
				}
			}
		}
		every_country = {
			limit = {
				is_in_faction_with = ROOT
				is_faction_leader = yes
			}
			country_event = { days = 1 id = germany_es.3 }
		}
	}
	option = { # Refuse
		name = germany_es.2.c
		ai_chance = { 
			factor = 10
			modifier = {
				factor = 0
				OR = {
					is_in_faction_with = GER
					AND = {
						is_historical_focus_on = yes
						ROOT = { tag = LIT }
					}
					AND = {
						ROOT = { tag = SOV }
						has_global_flag = sov_yes_pact
					}
				}
			} 
		}
		effect_tooltip = {
			GER = {
				create_wargoal = {
					type = take_state_focus
					target = ROOT
					generator = { 188 }
				}
				country_event = { days = 1 id = germany.23 }
			}
		}
		hidden_effect = {
			GER = {
				country_event = { days = 1 id = germany.23 }
			}
		}
	}
}
